makeblastdb -in DUX4_fusion_2A1.fa  -dbtype nucl -out ./seq3_blast/DUX_fusion_2A1.blastdb

makeblastdb -in DUX4_fusion_2A1.fa  -dbtype nucl -out ./seq3_blast/DUX_fusion_2A1_transcript.blastdb
